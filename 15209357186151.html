算法_最长递增子序列

2018/3/13 posted in  编程

问题：给定一个数组，求出该数组的最长递增子序列的长度以及个数

示例：

[1,3,5,4,7]
返回结果为(4，2) //其中长度为4，结果为2
其最长递增子序列分别为[1,3,5,7]和[1,3,4,7]

最终代码

/// 返回最长递增子序列的长度以及其个数
///
/// - Parameter data: 带统计的参数
/// - Returns: 最长递增子序列的长度以及其个数
func calculateLIS(data: [Int])-> (length: Int, count: Int) {
    
    /// 当前全局最长的递增子序列的长度
    var maxLength: Int = 0
    
    /// 存放以该index为结尾的最长递增子序列的长度
    var dp: [Int]  =  data.map { (_) -> Int in
        return 1
    }
    
    
    /// 存放以该index为结尾的最长递增子序列的个数
    var ans: [Int] = data.map { (_) -> Int in
        return 1
    }
    print("1111")
    data.enumerated().forEach { (index, value) in
        
        for i in 0..<index {
            print("\(index)   \(i)")
            if data[i] < data[index] {
                if dp[i] + 1 > dp[index] {
                    ans[index] = ans[i]
                    dp[index] = dp[i] + 1
                    maxLength = max(maxLength, dp[index])
                } else if dp[i] + 1 == dp[index] {
                    ans[index] += ans[i]
                }
            }
        }
    }
    
    let count = data.enumerated().filter { (index,value) -> Bool in
        return dp[index] == maxLength
        }.reduce(0) { (result, tuple) -> Int in
            return result + ans[tuple.offset]
    }
    
    return (maxLength,count)
}
